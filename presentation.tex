\section{Motivation}

\begin{frame}{RAG: Grounding LLM Outputs in Retrieved Evidence}
  \begin{columns}
    \column{0.56\textwidth}
    \textbf{Retrieval-Augmented Generation (RAG)}
    \begin{enumerate}\setlength{\itemsep}{0.4em}
      \item Retrieve $n$ documents $\mathbf{D}$ for query $q$
      \item Model generates $\hat{y} = \mathcal{M}(q, \mathbf{D})$
    \end{enumerate}

    \vspace{1em}
    \textbf{Key attribution question:}\\[0.4em]
    \textit{``Which documents actually caused\\
    the model to produce the correct answer?''}

    \vspace{1em}
    \textbf{Why it matters:}
    \begin{itemize}
      \item Citation and provenance
      \item Retrieval debugging
      \item User trust in high-stakes domains
    \end{itemize}

    \column{0.41\textwidth}
    \centering
    \begin{tikzpicture}[
      node distance=0.55cm,
      box/.style={draw,rounded corners,fill=uiucblue!10,
                  minimum width=3.0cm,minimum height=0.65cm,
                  font=\small},
      docbox/.style={draw,rounded corners,fill=conforange!25,
                     minimum width=3.0cm,minimum height=0.65cm,
                     font=\small},
      ansbox/.style={draw,rounded corners,fill=factgreen!30,
                     minimum width=3.0cm,minimum height=0.65cm,
                     font=\small}]
      \node[box]    (q)  {Query $q$};
      \node[box,below=of q]    (ret) {Retriever};
      \node[docbox,below=of ret]   (docs){$\mathbf{D} = \{d_1,\ldots,d_n\}$};
      \node[box,below=of docs]  (llm) {LLM $\mathcal{M}$};
      \node[ansbox,below=of llm] (ans) {Answer $\hat{y}$};
      \draw[->,thick] (q)--(ret);
      \draw[->,thick] (ret)--(docs);
      \draw[->,thick] (docs)--(llm);

    \end{tikzpicture}
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{Prior Work: Attribution as Causal Influence}
  \begin{columns}
    \column{0.56\textwidth}
    \textbf{Three leading attribution methods:}

    \vspace{0.6em}
    \begin{description}[leftmargin=0pt]\setlength{\itemsep}{0.6em}
      \item[\textbf{ContextCite}] Sparse linear surrogate (LASSO) over context ablations
      \item[\textbf{TracLLM}] Recursive partitioning + contribution denoising for long contexts
      \item[\textbf{AttriBoT}] Cached activations + proxy model for efficient leave-one-out
    \end{description}

    \vspace{0.8em}
    \textbf{Shared definition of importance:}
    \[
      d_i\ \text{is important} \Leftrightarrow
      \bigl|\dlogp(d_i)\bigr|\ \text{is large}
    \]

    \column{0.42\textwidth}
    \begin{alertblock}{Shared limitation}
      All three methods measure\\
      \textbf{causal influence on output probability}.\\[0.5em]
      None measure whether the\\
      output is \textbf{factually correct}.\\[0.5em]
      A document asserting a\\
      \textit{confidently wrong fact} would\\
      score as \textbf{highly important}\\
      under all three.
    \end{alertblock}
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{The Core Gap: Influence $\neq$ Usefulness}
  \begin{columns}
    \column{0.5\textwidth}

    \begin{tikzpicture}[scale=0.88,every node/.style={font=\small}]
      % Full context box
      \node[draw,fill=uiucblue!8,rounded corners,
            minimum width=5.6cm,minimum height=1.1cm,align=center]
        (full) at (0,3.0)
        {Full context $\mathbf{D}$\\
         \footnotesize Model resists, outputs \textcolor{factgreen}{\textbf{correct}} $\hat{y}$};

      % Doc removed
      \node[draw,fill=conforange!18,rounded corners,
            minimum width=5.6cm,minimum height=1.1cm,align=center]
        (abl) at (0,1.3)
        {Remove $d_i$ (counterfactual doc)\\
         \footnotesize Still outputs \textcolor{factgreen}{\textbf{correct}} $\hat{y}_{-i}$};

      \draw[->,thick] (full)--(abl) node[midway,right=2pt]
        {\footnotesize ablation};

      % Axes
      \node[draw,fill=conforange!30,rounded corners,
            minimum width=2.5cm,minimum height=0.7cm,align=center]
        at (-1.4,0.1)
        {\footnotesize $\dlogp \ll 0$\\[-0.1em]\tiny logprob collapses};

      \node[draw,fill=factgreen!20,rounded corners,
            minimum width=2.5cm,minimum height=0.7cm,align=center]
        at (1.4,0.1)
        {\footnotesize $\dfact = 0$\\[-0.1em]\tiny fact unchanged};

      \draw[->,dashed,thick,conforange] (abl.south west) -- (-1.4,0.45);
      \draw[->,dashed,thick,factgreen] (abl.south east) -- (1.4,0.45);

      % Verdict
      \node[draw,fill=uiucorange!20,rounded corners,
            minimum width=5.6cm,minimum height=0.7cm,align=center]
        at (0,-0.8)
        {\textbf{Attribution:} ``$d_i$ is important'' \quad
         \textbf{Reality:} factually useless};
    \end{tikzpicture}

    \column{0.47\textwidth}
    \textbf{What is happening?}

    \vspace{0.5em}
    The model \textit{anchors its confidence} on\\
    the counterfactual document $d_i$.\\[0.5em]
    When $d_i$ is removed:\\
    \begin{itemize}
      \item Model confidence in $\hat{y}$ \textbf{drops} (logprob collapses)
      \item But the \textbf{correct answer persists}
    \end{itemize}

    \vspace{0.8em}
    \begin{alertblock}{Collapse Divergence}
      Large $|\dlogp|$ \textbf{without} any $\dfact$\\
      Logprob signals importance.\\
      Factuality signals uselessness.\\
      They disagree.
    \end{alertblock}
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{Why This Matters}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{High-stakes RAG applications}

    \vspace{0.4em}
    \begin{itemize}\setlength{\itemsep}{0.5em}
      \item \textbf{Medicine:} ``Which document\\
            caused this drug dosage recommendation?''
      \item \textbf{Law:} ``Which source supports\\
            this legal conclusion?''
      \item \textbf{Finance:} ``Which filing caused\\
            this risk assessment?''
    \end{itemize}

    \vspace{0.6em}
    If attribution labels a \bad{confidence-only}\\
    document as ``important'' $\Rightarrow$\\[0.3em]
    \hilight{Wrong citation. Wrong trust. Wrong guidance.}

    \column{0.48\textwidth}
    \begin{exampleblock}{This paper}
      Extend attribution with factual alignment.\\[0.4em]
      Measure importance on \textbf{three axes}:
      \begin{enumerate}\setlength{\itemsep}{0.3em}
        \item Did removing $d_i$ flip factual correctness?
        \item Did removing $d_i$ collapse log-probability?
        \item Did removing $d_i$ shift internal representations?
      \end{enumerate}
    \end{exampleblock}

    \vspace{0.5em}
    \textbf{Claim:} Axis 1 ($\dfact$) is necessary.\\
    Axes 2--3 alone are insufficient.
  \end{columns}
\end{frame}

\section{Problem Formulation}
% ================================================================

\begin{frame}{Formal Setup and LODO Ablation}
  \begin{columns}
    \column{0.52\textwidth}
    \textbf{Notation}
    \begin{itemize}\setlength{\itemsep}{0.3em}
      \item Query: $q$
      \item Retrieved documents: $\mathbf{D} = \{d_1, \ldots, d_n\}$
      \item Model: $\mathcal{M}$, ground truth: $y^*$
      \item Baseline answer: $\hat{y} = \mathcal{M}(q, \mathbf{D})$
      \item Factuality scorer: $f(\hat{y}, y^*) \in \{0, 1\}$
    \end{itemize}

    \vspace{0.8em}
    \textbf{LODO ablation} for document $d_i$:
    \[
      \mathbf{D}_{-i} = \mathbf{D} \setminus \{d_i\}
    \]
    \[
      \hat{y}_{-i} = \mathcal{M}(q,\, \mathbf{D}_{-i})
    \]

    \column{0.46\textwidth}
    \textbf{Three degradation axes:}

    \vspace{0.5em}
    \textbf{(1) Factual degradation}
    \[
      \dfact(d_i) = f(\hat{y}, y^*) - f(\hat{y}_{-i}, y^*)
    \]

    \textbf{(2) Logprob collapse}
    \[
      \dlogp(d_i) = \log P(\hat{y}\,|\,\mathbf{D}_{-i})
                   - \log P(\hat{y}\,|\,\mathbf{D})
    \]
    \footnotesize Teacher-forcing over $\hat{y}$;\\
    fully deterministic, temperature-free.

    \normalsize
    \vspace{0.4em}
    \textbf{(3) Representation drift}
    \[
      \dltwo^{(\ell)}(d_i) =
      \bigl\lVert \bar{h}^{(\ell)}_{\mathbf{D}} -
      \bar{h}^{(\ell)}_{\mathbf{D}_{-i}} \bigr\rVert_2
    \]
    \footnotesize Mean hidden state at layer $\ell$.
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{What the Two Definitions of Importance Look Like}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{block}{Prior definition: logprob-important}
      \[
        d_i\ \text{is important} \Leftrightarrow \dlogp(d_i) < -2.0
      \]
      Measures: \textit{how much did model confidence shift?}\\[0.4em]
      Does \textbf{not} ask: \textit{was the answer right?}
    \end{block}

    \vspace{0.6em}
    \begin{exampleblock}{Our definition: factually useful}
      \[
        d_i\ \text{is useful} \Leftrightarrow \dfact(d_i) > 0
      \]
      Requires a \textbf{correct baseline}: $f(\hat{y}, y^*) = 1$\\[0.3em]
      Measures: \textit{did removing $d_i$ break a correct answer?}
    \end{exampleblock}

    \column{0.48\textwidth}
    \textbf{Collapse Divergence Rate (\cdr)}

    \vspace{0.4em}
    Among all logprob-important documents,\\
    what fraction provide \textbf{zero factual benefit}?

    \[
      \cdr = \frac{|\{d_i : \dlogp < -2,\ \dfact = 0\}|}
                  {|\{d_i : \dlogp < -2\}|}
    \]

    \vspace{0.4em}
    \begin{itemize}
      \item \cdr = 1.0 $\Rightarrow$ logprob-only attribution\\
            is entirely factually uninformative
      \item \cdr = 0.0 $\Rightarrow$ perfect alignment
    \end{itemize}

    \vspace{0.4em}
    \hilight{We find CDR $\approx$ 0.93}
  \end{columns}
\end{frame}

\begin{frame}{Factuality-Aware Importance Taxonomy}
  Classify every ablated document by its $(\dlogp,\, \dfact)$ pair:

  \vspace{0.6em}
  \begin{center}
  \renewcommand{\arraystretch}{1.35}
  \begin{tabular}{lllp{5cm}}
    \toprule
    Category & $\dlogp$ & $\dfact$ & Meaning \\
    \midrule
    \textcolor{helpblue}{\textbf{Helpful-important}}
      & $< -2$ & $> 0$
      & Causally \emph{and} factually useful\\
    \textcolor{conforange}{\textbf{Confidence-only}}
      & $< -2$ & $= 0$
      & Influential but \textbf{factually useless} (CDR population)\\
    \textcolor{uiucorange}{\textbf{Harmful-influential}}
      & $< -2$ & $< 0$
      & Influence pushes toward wrong answer\\
    \addlinespace
    \textcolor{factgreen}{\textbf{Fact-only}}
      & $\geq -2$ & $> 0$
      & Quietly necessary; modest logprob impact\\
    \textcolor{neutralgray}{\textbf{Neutral}}
      & $\geq -2$ & $= 0$
      & No measurable effect on either axis\\
    \textbf{Harmful-weak}
      & $\geq -2$ & $< 0$
      & Weakly degrades factuality\\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{Factuality-aware causal score:}
    \[
      s_{\mathrm{FA}}(d_i) = \max(0,-\dlogp) \cdot \max(0,\dfact)
    \]
    \column{0.48\textwidth}
    \footnotesize Non-zero only when a document is \textbf{both}
    logprob-important \emph{and} factually useful.\\
    Directly operationalizes ``influence aligned with correctness.''
  \end{columns}
\end{frame}

\begin{frame}{Why en\_counter\_mid?}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{RGB benchmark (Chen et al.\ 2024)}

    \vspace{0.4em}
    \begin{description}[leftmargin=0pt]\setlength{\itemsep}{0.4em}
      \item[\texttt{en\_mid}]
        Standard retrieval. Near-ceiling accuracy\\
        (92--96\%) regardless of noise level.\\
        \textit{Removing any single document\\
        rarely changes the output.}\\
        \bad{LODO is causally uninformative.}
      \item[\texttt{en\_counter\_mid}]
        All retrieved docs assert the \textbf{wrong} answer.\\
        Model must resist the evidence to be correct.\\
        Accuracy: 0--20\%.\\
        \good{Every document's presence is causally observable.}
    \end{description}

    \column{0.48\textwidth}
    \includegraphics[width=\linewidth]{CF4_dataset_motivation.png}
    \vspace{0.3em}
    \small Baseline accuracy comparison. \texttt{en\_mid} (near-ceiling)
    makes document removal a non-event. \texttt{en\_counter\_mid} creates
    the conditions where causal perturbation is measurable.
  \end{columns}
\end{frame}

\begin{frame}{RQ3 — Two Distinct Regimes}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{alertblock}{Context starvation ($n=3$)}
      \begin{itemize}
        \item Baseline accuracy = \textbf{0\%}
        \item Model never produces correct answer\\
              with 3 counterfactual docs
        \item Any removal $\Rightarrow$ logprob collapse\\
              (model has nothing to stand on)
        \item $\dfact$ cannot be $>0$ (no correct\\
              baseline to degrade from)
        \item Gap = 50\%; CDR = 93\%
      \end{itemize}
    \end{alertblock}

    \column{0.48\textwidth}
    \begin{exampleblock}{Redundancy regime ($n \geq 5$)}
      \begin{itemize}
        \item Baseline accuracy becomes non-zero
        \item Occasionally correct baselines exist\\
              $\Rightarrow$ factual degradation observable
        \item Remaining docs buffer each removal\\
              $\Rightarrow$ logprob-important rate falls
        \item But: CDR stays above 90\%
        \item More docs $\neq$ better factual grounding
      \end{itemize}
    \end{exampleblock}

    \vspace{0.4em}
    \textbf{Adding docs doesn't fix the gap.}\\
    It just adds more confidence-only noise.
  \end{columns}
\end{frame}

\section{Discussion \& Conclusion}
% ================================================================

\begin{frame}{Summary: All Five Research Questions}
  \begin{center}
  \renewcommand{\arraystretch}{1.45}
  \begin{tabular}{llp{5cm}l}
    \toprule
    RQ & Question (short) & Finding & Key Number \\
    \midrule
    1 & Does logprob overestimate? & 9/10 logprob-imp.\ docs are factually useless & \hilight{CDR\,=\,92.7\%} \\
    2 & Can we rank better? & Factuality-aware score is perfect at rank 1 & \hilight{P@1: 1.00 vs 0.50} \\
    3 & Does $n$ change the gap? & Gap persists; CDR stable at all passage counts & CDR $>$90\% at all $n$ \\
    4 & Where in the model? & Late-layer integration; helpful $\approx$ conf-only drift & Horizon: layers 18--20 \\
    5 & Which tokens? & Collapse on 1--2 factual tokens only & 48\% of tokens affected \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{block}{Central message}
    Influence-based attribution correctly measures what shifts model output.\\
    It does \textbf{not} measure whether that shift was factually beneficial.\\[0.3em]
    \textbf{$\dfact$ is necessary.} Neither $\dlogp$ nor $\dltwo$ can substitute for it.
  \end{block}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{Implications}
  \begin{columns}
    \column{0.5\textwidth}
    \textbf{For attribution systems}
    \begin{itemize}\setlength{\itemsep}{0.4em}
      \item ContextCite / TracLLM / AttriBoT:\\
            \textit{correct but incomplete}
      \item Extend: weight $\dlogp$ by $\dfact$ alignment
      \item Factuality-aware score as post-hoc filter
      \item Low-CDR corpus = more reliable RAG
    \end{itemize}

    \vspace{0.6em}
    \textbf{For RAG evaluation}
    \begin{itemize}\setlength{\itemsep}{0.4em}
      \item Track CDR as a corpus quality metric
      \item Diagnose retriever behavior:\\
            high CDR $\Rightarrow$ retriever is returning\\
            influential-but-useless documents
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Future directions}
    \begin{itemize}\setlength{\itemsep}{0.4em}
      \item \textbf{Position-controlled LODO}\\
            Isolate semantic absence from\\
            positional encoding artifact
      \item \textbf{Mid-layer signal (layer 18--20)}\\
            Real-time uncertainty detection\\
            before decoding completes
      \item \textbf{Factuality-aware reranking}\\
            Fine-tune retriever toward\\
            factually useful documents
      \item \textbf{More models and datasets}\\
            Generalize beyond Llama-3.1-8B\\
            and counterfactual setting
    \end{itemize}
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{Limitations}
  \begin{columns}
    \column{0.5\textwidth}
    \begin{alertblock}{Scope limitations}
      \begin{itemize}\setlength{\itemsep}{0.4em}
        \item Single model: Llama-3.1-8B-Instruct
        \item Single benchmark: RGB en\_counter\_mid\\
              (adversarial retrieval setting)
        \item CDR may differ in standard\\
              (non-counterfactual) retrieval
        \item Small scale: 10 queries,\\
              513 ablations total
      \end{itemize}
    \end{alertblock}

    \column{0.48\textwidth}
    \begin{alertblock}{Method limitations}
      \begin{itemize}\setlength{\itemsep}{0.4em}
        \item Post-hoc: requires correct-baseline queries;\\
              not usable at retrieval time
        \item $O(n)$ extra forward passes per query
        \item $\dfact$ depends on evaluator quality\\
              (exact-match may miss paraphrases)
        \item Token analysis: only 10 case studies\\
              (no AUROC over full ablation set)
        \item Positional encoding confound not\\
              yet controlled (E7 pending)
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

% ----------------------------------------------------------------
\begin{frame}{Conclusion}
  \begin{center}
    \LARGE\textbf{Influence Is Not Usefulness}
  \end{center}

  \vspace{0.4em}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\setlength{\itemsep}{0.55em}
      \item \textbf{LODO}: three-axis evaluation\\
            ($\dfact$, $\dlogp$, $\dltwo$) per removed document
      \item \textbf{CDR = 92.7\%}: among logprob-important\\
            documents, 9 out of 10 provide zero factual benefit
      \item \textbf{Six-category taxonomy}: separates\\
            helpful, confidence-only, and harmful influence
      \item \textbf{Factuality-aware ranking}: P@1 = 1.00\\
            vs 0.50 for logprob-only
      \item \textbf{Integration Horizon}: layers 18--20;\\
            logprob collapse localized to 1--2 factual tokens
      \item Influence-based attribution must be\\
            \textbf{extended with $\dfact$} to be reliable
    \end{itemize}

    \column{0.43\textwidth}
    \begin{center}
