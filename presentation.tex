\section{Motivation}
% ================================================================

\begin{frame}{Why Attribution Fails: Influence $\neq$ Usefulness}
  \begin{columns}
    \column{0.50\textwidth}
    \textbf{RAG attribution question:}\\[0.3em]
    \textit{``Which documents caused the correct answer?''}

    \vspace{0.7em}
    \textbf{Existing methods} (ContextCite, TracLLM, AttriBoT)\\
    define importance as:
    \[
      d_i\ \text{is important} \Leftrightarrow |\dlogp(d_i)|\ \text{is large}
    \]
    \textit{They measure confidence shift — not factual impact.}

    \vspace{0.7em}
    \begin{alertblock}{Collapse Divergence}
      A counter-factual document asserts the \textbf{wrong fact}.\\
      Removing it: logprob \textbf{collapses} ($\dlogp \ll 0$),\\
      but factual correctness \textbf{unchanged} ($\dfact = 0$).\\[0.3em]
      Attribution says ``important.'' Reality: factually irrelevant.
    \end{alertblock}

    \column{0.48\textwidth}
    \begin{tikzpicture}[scale=0.85,every node/.style={font=\small}]
      \node[draw,fill=uiucblue!8,rounded corners,
            minimum width=5.4cm,minimum height=1.0cm,align=center]
        (full) at (0,3.0)
        {Mixed context: 1 GT doc + $(n{-}1)$ CF docs\\
         \footnotesize Model outputs \textcolor{helpblue}{\textbf{correct}} $\hat{y}$};

      \node[draw,fill=conforange!18,rounded corners,
            minimum width=5.4cm,minimum height=1.0cm,align=center]
        (abl) at (0,1.3)
        {Remove \emph{counter-factual} doc $d_i$\\
         \footnotesize Still outputs \textcolor{helpblue}{\textbf{correct}} $\hat{y}_{-i}$};

      \draw[->,thick] (full)--(abl) node[midway,right=2pt]{\footnotesize ablation};

      \node[draw,fill=conforange!30,rounded corners,
            minimum width=2.3cm,minimum height=0.65cm,align=center]
        at (-1.35,0.1)
        {\footnotesize $\dlogp \ll 0$\\[-0.1em]\tiny logprob collapses};

      \node[draw,fill=neutralgray!30,rounded corners,
            minimum width=2.3cm,minimum height=0.65cm,align=center]
        at (1.35,0.1)
        {\footnotesize $\dfact = 0$\\[-0.1em]\tiny still correct};

      \draw[->,dashed,thick,conforange] (abl.south west) -- (-1.35,0.42);
      \draw[->,dashed,thick,neutralgray]  (abl.south east) -- (1.35,0.42);

      \node[draw,fill=uiucorange!20,rounded corners,
            minimum width=5.4cm,minimum height=0.65cm,align=center]
        at (0,-0.75)
        {\textbf{Attribution:} ``important'' \quad \textbf{Reality:} irrelevant};
    \end{tikzpicture}
  \end{columns}
\end{frame}

\section{Experiment Setup}
% ================================================================

\begin{frame}{Experiment Setup: Mixed Retrieval Setting}
  \begin{columns}
    \column{0.50\textwidth}
    \textbf{Dataset: Mixed retrieval from RGB}
    \begin{itemize}\setlength{\itemsep}{0.3em}
      \item Each query receives \textbf{1 ground-truth (GT) doc} from
            \texttt{en\_mid} positive + \textbf{$(n{-}1)$ counter-factual (CF) docs}
            from \texttt{en\_counter\_mid} positive
      \item CF docs are relevant but assert the \emph{wrong answer}
            (modified dates, wrong names)
      \item GT doc enables a non-zero baseline; majority context misleads
      \item Baseline accuracy: \textbf{20--30\%} (vs.\ 0--10\% in pure CF setting)
    \end{itemize}

    \vspace{0.4em}
    \textbf{Model \& Scale}
    \begin{itemize}\setlength{\itemsep}{0.3em}
      \item Llama-3.1-8B-Instruct, $\tau = 0.7$
      \item Passage sweep: $n \in \{3, 5, 7, 10\}$,\ 10 queries
      \item \textbf{198 sweep ablations} (473 total incl.\ en\_refine)
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Three evaluation axes per ablated document $d_i$:}

    \vspace{0.4em}
    \textbf{(1) Factual change}
    \[
      \dfact(d_i) = f(\hat{y}, y^*) - f(\hat{y}_{-i}, y^*)
    \]
    \textbf{(2) Logprob collapse}
    \[
      \dlogp(d_i) = \log P(\hat{y}\,|\,\mathbf{D}_{-i}) - \log P(\hat{y}\,|\,\mathbf{D})
    \]
    {\footnotesize Teacher-forcing over $\hat{y}$; fully deterministic.}

    \vspace{0.3em}
    \textbf{(3) Representation drift}
    \[
      \dltwo^{(\ell)}(d_i) = \bigl\lVert \bar{h}^{(\ell)}_{\mathbf{D}} - \bar{h}^{(\ell)}_{\mathbf{D}_{-i}} \bigr\rVert_2
    \]
    {\footnotesize Mean hidden state at layer $\ell$.}
  \end{columns}
\end{frame}

\section{Insights}
% ================================================================

\begin{frame}{Insights}
  \begin{columns}
    \column{0.50\textwidth}
    \textbf{Influence-usefulness gap across passage counts}

    \vspace{0.3em}
    \begin{tabular}{crrrrr}
      \toprule
      $n$ & base acc & logprob-imp & fact-imp & gap \\
      \midrule
      3  & 30\% & 70.0\% & 16.7\% & 53.3\% \\
      5  & 30\% & 38.6\% & 11.4\% & 27.3\% \\
      7  & 20\% & 30.9\% &  7.3\% & 23.6\% \\
      10 & 30\% & 18.8\% & 11.6\% &  7.2\% \\
      \bottomrule
    \end{tabular}

    \vspace{0.4em}
    \begin{itemize}\setlength{\itemsep}{0.3em}
      \item \textbf{Dilution effect}: as $n$ grows, GT doc's logprob collapse
            is diluted by more CF docs
      \item \textbf{Gap narrows with $n$}: logprob-imp rate drops faster than
            fact-impact rate
      \item \hilight{CDR 58--77\% at every $n$ — robust across scale}
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Where in the model? — A critical null finding}

    \vspace{0.3em}
    \includegraphics[width=\linewidth]{E4_layerwise_drift_by_taxonomy.png}

    \vspace{0.2em}
    \begin{itemize}\setlength{\itemsep}{0.3em}
      \item \textbf{Integration horizon: layers 18--20}
      \item \textbf{Factuality-critical $\approx$ Confidence-only in drift}\\
            L32: 15.68 vs 14.05 — statistically indistinguishable\\
            $\Rightarrow$ $\dltwo$ \emph{cannot} replace $\dfact$
      \item \textbf{The irreducible axis}: only $\dfact$ separates\\
            the truly useful document from the confident decoy
    \end{itemize}
  \end{columns}
\end{frame}

\section{Conclusion}
% ================================================================

\begin{frame}{Conclusion}
  \begin{center}
    \LARGE\textbf{Influence Is Not Usefulness}
  \end{center}

  \vspace{0.3em}
  \begin{columns}
    \column{0.55\textwidth}
    \begin{itemize}\setlength{\itemsep}{0.5em}
      \item \textbf{Mixed setting}: 1 GT doc among $(n{-}1)$ CF adversaries —
            mirrors realistic retrieval noise
      \item \textbf{CDR = 63.2\%}: majority of logprob-important docs are factually irrelevant
      \item \textbf{Doc-type asymmetry}:
            GT docs are factuality-critical 62.5\% of the time;
            CF docs are confidence-only 71.2\% — \emph{indistinguishable by logprob alone}
      \item \textbf{$\dfact$ is irreducible}: neither $\dlogp$ (P@1\,=\,0.73) nor
            $\dltwo$ (L32: 14--16 for both categories) can substitute
      \item \textbf{Factuality-aware ranking}: P@1\,=\,1.00, MRR\,=\,1.00 on 12 queries
      \item \textbf{Future work}: factuality-aware retriever reranking,
            mid-layer early-warning flag, position-controlled LODO, larger scale
    \end{itemize}
